fof(puro_model, axiom,
(
    % === Entities and Their PURO Categories ===
    associatedwith(bookstore_prague, bobject) &
    associatedwith(_someObjects1, bobject) &
    associatedwith(the_little_prince, btype_1) &
    associatedwith(book, btype_2) &
    associatedwith(price, btype_a) & 
    associatedwith(book_relationship, btype_r) &
    associatedwith(saint_exupery, bobject) &
    associatedwith(person, btype_1) & 
    

    % === B-Instantiations Using the Abbreviation x :B T ===
    binstanceof(_someObjects1, the_little_prince) &
    binstanceof(_someObjects1, someObjects) &
    binstanceof(the_little_prince, book) &
    binstanceof(bookstore_prague, bookstore) &
    binstanceof(saint_exupery, person) &
    binstanceof(author, book_relationship) &
    binstanceof(unit_price, price) &

  
    %=== B-Subsumption ====
   associatedwith(organization, btype_1) & 
   associatedwith(bookstore, btype_1) &
   bsubsume(bookstore, organization) &


    % === The Offers Relation ===
    associatedwith(offers, brelation) &
    relrole(_offersRole1) &
    relrole(what) &
    _offersRole1 != what &

    % === An Instance of the Offers B-Relationship ===
    associatedwith(_offers1, brelationship) &
    binstanceof(_offers1, offers) &
    hasParticipant(_offers1, _offersRole1, bookstore_prague) &
    hasParticipant(_offers1, what, _someObjects1) &

    %==== The Author Relation===
    associatedwith(author, brelation) &
    relrole(author_role) &
    relrole(authored_work_role) &
    author_role != authored_work_role &

    % === An Instance of the Author B-Relationship ===
    associatedwith(authorship_1, brelationship) &
    binstanceof(authorship_1, author) &
    hasParticipant(authorship_1, author_role, saint_exupery) & 
    hasParticipant(authorship_1, authored_work_role, the_little_prince) & 
    

    % === B-Attributes and Valuations ===
    associatedwith(quantity, battribute) &
    associatedwith(unit_price, battribute) &
    associatedwith(pages, battribute) &
    associatedwith(_quantity1, bvaluation) &
    associatedwith(_unit_price1, bvaluation) &
    associatedwith(_pages1, bvaluation) &
    binstanceof(_quantity1, quantity) &
    binstanceof(unit_price, price) &
    binstanceof(_unit_price1, unit_price) &
    binstanceof(_pages1, pages) &
    binstanceof(the_little_prince1, the_little_prince) &
    hasValuation(_someObjects1, _quantity1) &
    hasValue(_quantity1, 10) &
    hasValuation(_offers1, _unit_price1) &
    hasValue(_unit_price1, 300) &
    hasValuation(the_little_prince_1, _pages1) &
    hasValue(_pages1, 100)
)).
